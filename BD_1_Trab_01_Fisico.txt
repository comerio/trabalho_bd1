CREATE TABLE Usuario (
    Nome_Sobrenome VARCHAR(80),
    Sexo CHAR,
    Data_Nascimento DATE,
    RG INT,
    CPF INT,
    ID INT PRIMARY KEY,
    Senha VARCHAR(20),
    Nome_Usuario VARCHAR(80)
);

CREATE TABLE Animal (
    ID_Animal INT PRIMARY KEY,
    Nome VARCHAR(80),
    Sexo CHAR
);

CREATE TABLE Documentacao (
    Descricao VARCHAR(80),
    Data_Nascimento DATE,
    Vacinacao VARCHAR(80),
    ID_animal INT
);

CREATE TABLE Protector (
    ID_Protector INT PRIMARY KEY
);

CREATE TABLE Adopter (
    ID_Adopter INT PRIMARY KEY
);

CREATE TABLE Classifica (
    ID_Usuario_1 INT,
    ID_Usuario_2 INT,
    Nota INT
);

CREATE TABLE Cadastra (
    ID_animal INT,
    Protector_ID INT,
    Data_Cadastro DATE,
    Codigo_Cadastro INT PRIMARY KEY
);

CREATE TABLE Adota (
    ID_Animal INT,
    ID_Usuario INT,
    Data_Adocao DATE,
    Codigo_Adocao INT PRIMARY KEY
);

CREATE TABLE Agenda_Visita (
    ID_Protector INT,
    ID_Adopter INT,
    Data_Agendamento DATE,
    Codigo_Agendamento INT PRIMARY KEY
);

CREATE TABLE Dados_Contato (
    Num_Telefone INT,
    Endereco VARCHAR(80),
    Email VARCHAR(100),
    ID_Usuario INT
);
